def agrupa_valor_tabla(
    lista1: Optional[List[float]] = None,
    lista2: Optional[List[float]] = None,
    lista3: Optional[List[float]] = None,
    lista4: Optional[List[float]] = None
) -> List[float]:
    """
    Suma los elementos de hasta cuatro listas numricas y devuelve los resultados individuales.

    Parmetros:
    ----------
    lista1 : Optional[List[float]], opcional
        Primera lista de nmeros. Por defecto None.
    lista2 : Optional[List[float]], opcional
        Segunda lista de nmeros. Por defecto None.
    lista3 : Optional[List[float]], opcional
        Tercera lista de nmeros. Por defecto None.
    lista4 : Optional[List[float]], opcional
        Cuarta lista de nmeros. Por defecto None.

    Retorna:
    -------
    List[float]
        Lista con la suma de cada lista individual (en orden).
    """
    resultados = []

    for lista in (lista1, lista2, lista3, lista4):
        if lista is not None:
            resultados.append(sum(lista))
        else:
            resultados.append(0.0)

    return resultados

def agrupa_columna_tabla(
    lista1: Optional[List] = None,
    lista2: Optional[List] = None,
    lista3: Optional[List] = None
) -> List[Tuple]:
    """
    Agrupa dinmicamente elementos de las listas proporcionadas en una lista de tuplas.

    Solo se agrupan aquellas listas que se pasen explcitamente como argumento.
    Las listas deben tener la misma longitud para evitar prdida de datos, ya que se recortan
    al tamao de la ms corta (comportamiento de zip).

    Parmetros:
    ----------
    lista1 : Optional[List], opcional
        Primera columna de datos. Por defecto None.
    lista2 : Optional[List], opcional
        Segunda columna de datos. Por defecto None.
    lista3 : Optional[List], opcional
        Tercera columna de datos. Por defecto None.

    Retorna:
    -------
    List[Tuple]
        Lista de tuplas, donde cada tupla representa una fila con los elementos agrupados
        de las listas proporcionadas.
    """
    columnas: List[List] = []

    if lista1 is not None:
        columnas.append(lista1)
    if lista2 is not None:
        columnas.append(lista2)
    if lista3 is not None:
        columnas.append(lista3)

    return list(zip(*columnas)) if columnas else []

def obtiene_categoria(lista: Optional[List[Any]] = None) -> List[Any]:
    """
    Devuelve una lista con los elementos nicos de la lista proporcionada.

    Parmetros:
    ----------
    lista : Optional[List[Any]], opcional
        Lista de elementos (cualquier tipo). Si no se proporciona, se usa una lista vaca.

    Retorna:
    -------
    List[Any]
        Lista con los elementos nicos (sin repeticiones) de la lista original.
    """
    if lista is None:
        lista = []
    return list(set(lista))

def suma_venta_producto_categoria(
    lista1: Optional[List[str]] = None,
    lista2: Optional[List[Tuple[str, int]]] = None
) -> Dict[str, int]:
    """
    Crea un diccionario y suma los elementos de una lista de tuplas segn las claves presentes en lista1.

    Parmetros:
    ----------
    lista1 : Optional[List[str]], opcional
        Lista de claves (categoras) como strings. Por defecto None.
    lista2 : Optional[List[Tuple[str, int]]], opcional
        Lista de tuplas con pares (categora, valor). Por defecto None.

    Retorna:
    -------
    Dict[str, int]
        Diccionario con cada categora de lista1 como clave y la suma de los valores asociados
        en lista2. El diccionario resultante est ordenado de forma descendente por valor.
    """
    if lista1 is None:
        lista1 = []
    if lista2 is None:
        lista2 = []

    diccionario = {
        lista1[e]: sum([valor[1] for valor in lista2 if valor[0] == lista1[e]])
        for e in range(len(lista1))
    }

    return dict(sorted(diccionario.items(), key=lambda x: x[1], reverse=True))

def promedio(lista: List[float] = None) -> float:
    """
    Calcula el promedio de una lista de nmeros.

    Parmetros:
    ----------
    lista : List[float]
        Lista de valores numricos. Por defecto None.

    Retorna:
    -------
    float
        Promedio redondeado a 4 decimales. Si la lista est vaca o es None, retorna 0.0.
    """
    if lista is None:
      return 0.0
    return round(sum(lista) / len(lista), 4)

def muestra_venta(
    no_tienda: int = 0,
    diccionario: Optional[Dict[str, int]] = None,
    cantidad: int = 1,
    mas_vendido: bool = True
) -> None:
    """
    Muestra los productos ms o menos vendidos de una tienda.

    Parmetros:
    ----------
    no_tienda : int
        Nmero de la tienda (por defecto 0).
    diccionario : dict[str, int]
        Diccionario con productos como claves y ventas como valores. Por defecto None.
    cantidad : int, opcional
        Nmero de productos a mostrar (por defecto 1).
    mas_vendido : bool, opcional
        Si True muestra los ms vendidos, si False los menos vendidos (por defecto True).
    """
    if diccionario is None or not diccionario:
        print(f"No hay datos de venta disponibles para la tienda {no_tienda}.")
        return

    claves = list(diccionario.keys())
    valores = list(diccionario.values())

    cantidad = min(cantidad, len(diccionario))  # Evita IndexError

    if mas_vendido:
        for e in range(cantidad):
            print(f"El producto No.{e + 1} ms vendido de la tienda {no_tienda} es: {claves[e]} y la cantidad vendida fue: {valores[e]}")
    else:
        for e in range(1, cantidad + 1):
            print(f"El producto No.{e} menos vendido de la tienda {no_tienda} es: {claves[-e]} y la cantidad vendida fue: {valores[-e]}")

def obtiene_eje_x(
    diccionario: Optional[Dict[str, int]] = None,
    lista: Optional[List[str]] = None,
    numero: int = 0
) -> List[int]:
    """
    Obtiene los valores de un diccionario segn un orden dado y prepara una lista para graficar.

    Parmetros:
    ----------
    diccionario : dict[str, int], opcional
        Diccionario con categoras como claves y cantidades como valores.
    lista : list[str], opcional
        Lista que define el orden en el que se graficarn las categoras.
    numero : int, opcional
        Cantidad de categoras a mostrar con sus valores reales; el resto se rellena con ceros.

    Retorna:
    -------
    list[int]
        Lista de valores a graficar, ordenados segn `lista`, con ceros donde no se desea mostrar el valor.
    """
    if diccionario is None:
        diccionario = {}
    if lista is None:
        lista = []

    eje_x = []

    for elemento in lista:
        if elemento in diccionario:
            # Obtener la posicin del elemento en el diccionario original
            pos = list(diccionario.keys()).index(elemento)
            # Si est dentro del rango deseado, se agrega su valor; si no, se agrega 0
            valor = diccionario[elemento] if pos < numero else 0
            eje_x.append(valor)
        else:
            # Si el elemento no est en el diccionario, se asume como 0
            eje_x.append(0)

    return eje_x

def obtiene_ejes(
    diccionario: Optional[Dict[str, int]] = None,
    cantidad: int = 0,
    mas_vendido: bool = True
) -> Tuple[List[str], List[int]]:
    """
    Obtiene listas de claves y valores desde un diccionario ordenado, para graficar productos ms o menos vendidos.

    Parmetros:
    ----------
    diccionario : dict[str, int], opcional
        Diccionario con productos como claves y cantidades vendidas como valores.
    cantidad : int, opcional
        Nmero de productos a mostrar (por defecto 0).
    mas_vendido : bool, opcional
        Si True muestra los ms vendidos; si False, los menos vendidos (por defecto True).

    Retorna:
    -------
    tuple[list[str], list[int]]
        Dos listas: eje X con los nombres de productos y eje Y con las cantidades correspondientes.
    """
    if diccionario is None:
        diccionario = {}

    # Obtener lista de claves y valores segn el orden
    claves = list(diccionario.keys())
    valores = list(diccionario.values())

    # Si se piden menos vendidos, invertir listas
    if not mas_vendido:
        claves.reverse()
        valores.reverse()

    # Recortar listas segn la cantidad solicitada
    eje_x = claves[:cantidad]
    eje_y = valores[:cantidad]

    return eje_x, eje_y
